\email{\VAR{email}}
\BLOCK{if website}\homepage{\VAR{website}}\BLOCK{endif}
\BLOCK{if github}\github{\VAR{github}}\BLOCK{endif}
\BLOCK{if linkedin}\linkedin{\VAR{linkedin}}\BLOCK{endif}

%-------------------------------------------------------------------------------
%	LETTER INFORMATION

\recipient
  {\VAR{cover.recipient}}
  {\BLOCK{for line in cover.address}\BLOCK{if not loop.first}\\\BLOCK{endif}\VAR{line}\BLOCK{endfor}}

\lettertitle{\VAR{cover.title}}
% How the letter is opened
\letteropening{\VAR{cover.opening}}
% How the letter is closed
\letterclosing{\VAR{cover.closing}}
% Any enclosures with the letter
\BLOCK{if cover.attachments}
\letterenclosure[Attached]{\BLOCK{for line in cover.attachments}\BLOCK{if not loop.first}, \BLOCK{endif}\VAR{line}\BLOCK{endfor}}
\BLOCK{endif}


%-------------------------------------------------------------------------------
\begin{document}

% Print the header with above personal informations
% Give optional argument to change alignment(C: center, L: left, R: right)

\makeletterclosing
